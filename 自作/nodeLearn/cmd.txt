中间件

express.use(function(req, res, next)
{
    var body = '';
    req.on('data', function(data){ body += data });
    req.on('end', function(data){ req.body = body; next(); })
})
express.use('/login', function(req, res, next)
{
   console.log(req.body);  //此处的req 保留上次修改
})

const cookieParser = require(cookie-parser);
server.use(cookieParser(res.secret));
server.use( '/', function(req, res)
{
    req.cookies;              //未签名cookies
